78 inline Eigen::Vector2d ClosestPointOnLineSegment2D(const Eigen::Ref<const Eigen::Vector2d> &point,
79  const Eigen::Ref<const Eigen::Vector2d> &line_start,
80  const Eigen::Ref<const Eigen::Vector2d> &line_end) {
81  Eigen::Vector2d diff = line_end - line_start;
82  double sq_norm = diff.squaredNorm();
83 
84  if (sq_norm == 0){
85  return line_start;
86  }
87 
88  double u = ((point.x() - line_start.x()) * diff.x() + (point.y() - line_start.y()) * diff.y()) / sq_norm;
89 
90  if (u <= 0) {
91  return line_start;
92  } else if (u >= 1) {
93  return line_end;
94  }
95 
96  return line_start + u * diff;
97 }

122 inline bool CheckLineSegmentsIntersection2D(const Eigen::Ref<const Eigen::Vector2d> &line1_start,
123  const Eigen::Ref<const Eigen::Vector2d> &line1_end,
124  const Eigen::Ref<const Eigen::Vector2d> &line2_start,
125  const Eigen::Ref<const Eigen::Vector2d> &line2_end,
126  Eigen::Vector2d *intersection = NULL) {
127  double s_numer, t_numer, denom, t;
128  Eigen::Vector2d line1 = line1_end - line1_start;
129  Eigen::Vector2d line2 = line2_end - line2_start;
130 
131  denom = line1.x() * line2.y() - line2.x() * line1.y();
132  if (denom == 0) {
133  return false;
134  }
135  bool denomPositive = denom > 0;
136 
137  Eigen::Vector2d aux = line1_start - line2_start;
138 
139  s_numer = line1.x() * aux.y() - line1.y() * aux.x();
140  if ((s_numer < 0) == denomPositive) {
141  return false;
142  }
143 
144  t_numer = line2.x() * aux.y() - line2.y() * aux.x();
145  if ((t_numer < 0) == denomPositive) {
146  return false;
147  }
148 
149  if (((s_numer > denom) == denomPositive) || ((t_numer > denom) == denomPositive)) {
150  return false;
151  }
152 
153  t = t_numer / denom;
154  if (intersection) {
155  *intersection = line1_start + t * line1;
156  }
157 
158  return true;
159 }

170 inline double DistanceSegmentToSegment2D(const Eigen::Ref<const Eigen::Vector2d> &line1_start,
171  const Eigen::Ref<const Eigen::Vector2d> &line1_end,
172  const Eigen::Ref<const Eigen::Vector2d> &line2_start,
173  const Eigen::Ref<const Eigen::Vector2d> &line2_end) {
174 
175  if (CheckLineSegmentsIntersection2D(line1_start, line1_end, line2_start, line2_end)){
176  return 0;
177  }
178 
179  std::array<double, 4> distances;
180 
181  distances[0] = DistancePointToSegment2D(line1_start, line2_start, line2_end);
182  distances[1] = DistancePointToSegment2D(line1_end, line2_start, line2_end);
183  distances[2] = DistancePointToSegment2D(line2_start, line1_start, line1_end);
184  distances[3] = DistancePointToSegment2D(line2_end, line1_start, line1_end);
185 
186  return *std::min_element(distances.begin(), distances.end());
187 }
