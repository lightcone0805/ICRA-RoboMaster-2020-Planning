59  static void CalculateMinAndMaxDistances (const std::vector<Eigen::Vector2d>& footprint, double& min_dist, double& max_dist) {
60  min_dist = std::numeric_limits<double>::max();
61  max_dist = 0.0;
62 
63  if (footprint.size() <= 2)
64  {
65  return;
66  }
67 
68  for (unsigned int i = 0; i < footprint.size() - 1; ++i) {
69  double vertex_dist = Distance (0.0, 0.0, footprint[i].coeffRef(0), footprint[i].coeffRef(1));
70  double edge_dist = PointToLineDistance(0.0, 0.0, footprint[i].coeffRef(0), footprint[i].coeffRef(1),
71  footprint[i + 1].coeffRef(0), footprint[i + 1].coeffRef(1));
72  min_dist = std::min(min_dist, std::min(vertex_dist, edge_dist));
73  max_dist = std::max(max_dist, std::max(vertex_dist, edge_dist));
74  }
75 
76  double vertex_dist = Distance(0.0, 0.0, footprint.back().coeffRef(0), footprint.back().coeffRef(1));
77  double edge_dist = PointToLineDistance(0.0, 0.0, footprint.back().coeffRef(0), footprint.back().coeffRef(1),
78  footprint.front().coeffRef(0), footprint.front().coeffRef(1));
79  min_dist = std::min(min_dist, std::min(vertex_dist, edge_dist));
80  max_dist = std::max(max_dist, std::max(vertex_dist, edge_dist));
81  }

135 RobotFootprintModelPtr GetRobotFootprintModel (T config) {
136  if (!config.robot_type(0).has_type()) {
137  ROS_INFO("No robot footprint model specified. Using point-shaped model.");
138  return boost::make_shared<PointRobotFootprint>();
139  } else if (config.robot_type(0).type() == 0) { // point type
140  ROS_INFO("Footprint model 'point' loaded");
141  return boost::make_shared<PointRobotFootprint>();
142  } else if (config.robot_type(0).type() == 1) {// circular
143  // get radius
144  double radius;
145  if (!config.robot_type(0).has_radius()) {
146  ROS_INFO("the circle radius doesn't exist, Using point-shaped model");
147  return boost::make_shared<PointRobotFootprint>();
148  }
149  radius = config.robot_type(0).radius();
150  ROS_INFO("Footprint model 'circular' (radius: %f m) loaded", radius);
151  return boost::make_shared<CircularRobotFootprint>(radius);
152  } else if (config.robot_type(0).type() == 3) { // line
153  if (! config.robot_type(0).robot_vertices_size()) {
154  ROS_INFO("the line vertices don't exist, Using point-shaped model");
155  return boost::make_shared<PointRobotFootprint>();
156  }
157  // get line coordinates
158  Eigen::Vector2d line_start, line_end;
159  line_start.coeffRef(0) = config.robot_type(0).robot_vertices(0).x();
160  line_start.coeffRef(1) = config.robot_type(0).robot_vertices(0).y();
161 
162  line_end.coeffRef(0) = config.robot_type(0).robot_vertices(1).x();
163  line_end.coeffRef(1) = config.robot_type(0).robot_vertices(1).y();
164 
165  if (config.robot_type(0).robot_vertices_size() != 2) {
166  ROS_INFO("the vertices' size doesn't equal 2, Using point-shaped model");
167  return boost::make_shared<PointRobotFootprint>();
168  }
169 
170  ROS_INFO("Footprint model 'line' loaded");
171  return boost::make_shared<LineRobotFootprint>(line_start, line_end);
172  } else if (config.robot_type(0).type() == 2) {
173 
174  if (!config.robot_type(0).front_offset()|| !config.robot_type(0).front_radius()
175  || !config.robot_type(0).rear_offset() || !config.robot_type(0).rear_radius() ) {
176  ROS_INFO("the param is not enough to combined two circle, Using point-shaped model");
177  return boost::make_shared<PointRobotFootprint>();
178  }
179  double front_offset, front_radius, rear_offset, rear_radius;
180  front_offset = config.robot_type(0).front_offset();
181  front_radius = config.robot_type(0).front_radius();
182  rear_offset = config.robot_type(0).rear_offset();
183  rear_radius = config.robot_type(0).rear_radius();
184 
185  ROS_INFO("Footprint model 'two circle' loaded");
186  return boost::make_shared<TwoCirclesRobotFootprint>(front_offset, front_radius, rear_offset, rear_radius);
187  } else if ( config.robot_type(0).type() == 4) {
188 
189  if (!config.robot_type(0).robot_vertices_size() ) {
190  ROS_INFO("the vertices is null, Using point-shaped model");
191  return boost::make_shared<PointRobotFootprint>();
192  }
193  Point2dContainer polygon;
194  for (int j = 0; j < config.robot_type(0).robot_vertices_size(); ++j) {
195  Eigen::Vector2d temp_point;
196  temp_point.coeffRef(0) = config.robot_type(0).robot_vertices(j).x();
197  temp_point.coeffRef(1) = config.robot_type(0).robot_vertices(j).y();
198  polygon.push_back(temp_point);
199  }
200 
201  ROS_INFO("Footprint model 'polygon' loaded");
202  return boost::make_shared<PolygonRobotFootprint>(polygon);
203  } else {
204  ROS_INFO("this type doesn't exist, Using point-shaped model");
205  return boost::make_shared<PointRobotFootprint>();
206  }
207 
208 }
