61 bool TebVertexConsole::InitTEBtoGoal(BidirIter path_start,
62  BidirIter path_end,
63  Fun fun_position,
64  double max_vel_x,
65  double max_vel_theta,
66  boost::optional<double> max_acc_x,
67  boost::optional<double> max_acc_theta,
68  boost::optional<double> start_orientation,
69  boost::optional<double> goal_orientation,
70  int min_samples,
71  bool guess_backwards_motion) {
72  Eigen::Vector2d start_position = fun_position(*path_start);
73  Eigen::Vector2d goal_position = fun_position(*boost::prior(path_end));
74 
75  bool backwards = false;
76 
77  double start_orient, goal_orient;
78  if (start_orientation) {
79  start_orient = *start_orientation;
80 
81  if (guess_backwards_motion
82  && (goal_position - start_position).dot(Eigen::Vector2d(std::cos(start_orient), std::sin(start_orient))) < 0)
83  backwards = true;
84  } else {
85  Eigen::Vector2d start2goal = goal_position - start_position;
86  start_orient = atan2(start2goal[1], start2goal[0]);
87  }
88  double timestep = 1;
89 
90 
91  if (goal_orientation) {
92  goal_orient = *goal_orientation;
93  } else {
94  goal_orient = start_orient;
95  }
96 
97  if (!IsInit()) {
98  AddPose(start_position, start_orient, true);
99 
100  std::advance(path_start, 1);
101  std::advance(path_end, -1);
102  int idx = 0;
103  for (; path_start != path_end; ++path_start)
104  {
105  Eigen::Vector2d curr_point = fun_position(*path_start);
106  Eigen::Vector2d
107  diff_last = curr_point - Pose(idx).GetPosition();
108  double diff_norm = diff_last.norm();
109 
110  double timestep_vel = diff_norm / max_vel_x;
111  double timestep_acc;
112  if (max_acc_x) {
113  timestep_acc = sqrt(2 * diff_norm / (*max_acc_x));
114  if (timestep_vel < timestep_acc && max_acc_x) timestep = timestep_acc;
115  else timestep = timestep_vel;
116  } else timestep = timestep_vel;
117 
118  if (timestep < 0) timestep = 0.2;
119 
120  double yaw = atan2(diff_last[1], diff_last[0]);
121  if (backwards)
122  yaw = g2o::normalize_theta(yaw + M_PI);
123  AddPoseAndTimeDiff(curr_point, yaw, timestep);
124 
125  Eigen::Vector2d diff_next = fun_position(*boost::next(path_start))
126  - curr_point;
127  double ang_diff = std::abs(g2o::normalize_theta(atan2(diff_next[1], diff_next[0])
128  - atan2(diff_last[1], diff_last[0])));
129 
130  timestep_vel = ang_diff / max_vel_theta;
131  if (max_acc_theta) {
132  timestep_acc = sqrt(2 * ang_diff / (*max_acc_theta));
133  if (timestep_vel < timestep_acc) timestep = timestep_acc;
134  else timestep = timestep_vel;
135  } else timestep = timestep_vel;
136 
137  if (timestep < 0) timestep = 0.2;
138 
139  yaw = atan2(diff_last[1], diff_last[0]);
140  if (backwards)
141  yaw = g2o::normalize_theta(yaw + M_PI);
142  AddPoseAndTimeDiff(curr_point, yaw, timestep);
143 
144  ++idx;
145  }
146  Eigen::Vector2d diff = goal_position - Pose(idx).GetPosition();
147  double diff_norm = diff.norm();
148  double timestep_vel = diff_norm / max_vel_x;
149  if (max_acc_x) {
150  double timestep_acc = sqrt(2 * diff_norm / (*max_acc_x));
151  if (timestep_vel < timestep_acc) timestep = timestep_acc;
152  else timestep = timestep_vel;
153  } else timestep = timestep_vel;
154 
155  DataBase goal(goal_position, goal_orient);
156 
157 
158  if (SizePoses() < min_samples - 1) {
159  while (SizePoses() < min_samples - 1) {
160 
161  DataBase temp_data;
162  temp_data.AverageInPlace(BackPose(), goal);
163 
164  AddPoseAndTimeDiff(temp_data, timestep);
165  }
166  }
167 
168  AddPoseAndTimeDiff(goal, timestep);
169  SetPoseVertexFixed(SizePoses() - 1, true);
170  } else {
171  return false;
172  }
173  return true;
174 }
