167  void RotateGlobal (double angle, bool adjust_theta=true) {
168  double new_x = std::cos(angle)*position_.x() - std::sin(angle)*position_.y();
169  double new_y = std::sin(angle)*position_.x() + std::cos(angle)*position_.y();
170  position_.x() = new_x;
171  position_.y() = new_y;
172  if (adjust_theta)
173  theta_ = g2o::normalize_theta(theta_+angle);
174  }
