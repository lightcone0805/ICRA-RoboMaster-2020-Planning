53 inline double PointToLineDistance (double px, double py, double x0, double y0, double x1, double y1) {
54  double A = px - x0;
55  double B = py - y0;
56  double C = x1 - x0;
57  double D = y1 - y0;
58 
59  double dot = A * C + B * D;
60  double len_sq = C * C + D * D;
61  double param = dot / len_sq;
62 
63  double xx, yy;
64 
65  if (param < 0)
66  {
67  xx = x0;
68  yy = y0;
69  }
70  else if (param > 1)
71  {
72  xx = x1;
73  yy = y1;
74  }
75  else
76  {
77  xx = x0 + param * C;
78  yy = y0 + param * D;
79  }
80 
81  return Distance(px, py, xx, yy);
82 }
