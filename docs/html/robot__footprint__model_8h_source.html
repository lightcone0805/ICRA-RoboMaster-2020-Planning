151 class TwoCirclesRobotFootprint : public BaseRobotFootprintModel {
152  public:
153 
154  TwoCirclesRobotFootprint(double front_offset, double front_radius, double rear_offset, double rear_radius)
155  : front_offset_(front_offset),
156  front_radius_(front_radius),
157  rear_offset_(rear_offset),
158  rear_radius_(rear_radius) {}
159 
160  virtual ~TwoCirclesRobotFootprint() {}
161 
162  void SetParameters(double front_offset, double front_radius, double rear_offset, double rear_radius) {
163  front_offset_ = front_offset;
164  front_radius_ = front_radius;
165  rear_offset_ = rear_offset;
166  rear_radius_ = rear_radius;
167  }
168 
169  virtual double CalculateDistance(const DataBase &current_pose, const Obstacle *obstacle) const {
170  Eigen::Vector2d dir = current_pose.OrientationUnitVec();
171  double dist_front = obstacle->GetMinimumDistance(current_pose.GetPosition() + front_offset_ * dir) - front_radius_;
172  double dist_rear = obstacle->GetMinimumDistance(current_pose.GetPosition() - rear_offset_ * dir) - rear_radius_;
173  return std::min(dist_front, dist_rear);
174  }
175 
176  virtual double GetInscribedRadius() {
177  double min_longitudinal = std::min(rear_offset_ + rear_radius_, front_offset_ + front_radius_);
178  double min_lateral = std::min(rear_radius_, front_radius_);
179  return std::min(min_longitudinal, min_lateral);
180  }
181 
182  private:
183 
184  double front_offset_;
185  double front_radius_;
186  double rear_offset_;
187  double rear_radius_;
188 
189 };

219  virtual double CalculateDistance(const DataBase &current_pose, const Obstacle *obstacle) const {
220  // here we are doing the transformation into the world frame manually
221  double cos_th = std::cos(current_pose.GetTheta());
222  double sin_th = std::sin(current_pose.GetTheta());
223  Eigen::Vector2d line_start_world;
224  line_start_world.x() = current_pose.GetPosition().coeffRef(0) + cos_th * line_start_.x() - sin_th * line_start_.y();
225  line_start_world.y() = current_pose.GetPosition().coeffRef(1) + sin_th * line_start_.x() + cos_th * line_start_.y();
226  Eigen::Vector2d line_end_world;
227  line_end_world.x() = current_pose.GetPosition().coeffRef(0) + cos_th * line_end_.x() - sin_th * line_end_.y();
228  line_end_world.y() = current_pose.GetPosition().coeffRef(1) + sin_th * line_end_.x() + cos_th * line_end_.y();
229  return obstacle->GetMinimumDistance(line_start_world, line_end_world);
230  }
231 
232  virtual double GetInscribedRadius() {
233  return 0.0; // lateral distance = 0.0
234  }
235 
236  private:
237 
238  Eigen::Vector2d line_start_;
239  Eigen::Vector2d line_end_;
240

242  EIGEN_MAKE_ALIGNED_OPERATOR_NEW

249 class PolygonRobotFootprint : public BaseRobotFootprintModel {
250  public:
251 
252  PolygonRobotFootprint(const Point2dContainer &vertices) : vertices_(vertices) {}
253 
254  virtual ~PolygonRobotFootprint() {}
255 
256  void SetVertices(const Point2dContainer &vertices) { vertices_ = vertices; }
257 
258  virtual double CalculateDistance(const DataBase &current_pose, const Obstacle *obstacle) const {
259  // here we are doing the transformation into the world frame manually
260  double cos_th = std::cos(current_pose.GetTheta());
261  double sin_th = std::sin(current_pose.GetTheta());
262 
263  Point2dContainer polygon_world(vertices_.size());
264  for (std::size_t i = 0; i < vertices_.size(); ++i) {
265  polygon_world[i].x() = current_pose.GetPosition().coeffRef(0) + cos_th * vertices_[i].x() - sin_th * vertices_[i].y();
266  polygon_world[i].y() = current_pose.GetPosition().coeffRef(1) + sin_th * vertices_[i].x() + cos_th * vertices_[i].y();
267 
268  }
269  return obstacle->GetMinimumDistance(polygon_world);
270  }
271 
272  virtual double GetInscribedRadius() {
273  double min_dist = std::numeric_limits<double>::max();
274  Eigen::Vector2d center(0.0, 0.0);
275 
276  if (vertices_.size() <= 2)
277  return 0.0;
278 
279  for (int i = 0; i < (int) vertices_.size() - 1; ++i) {
280  // compute distance from the robot center point to the first vertex
281  double vertex_dist = vertices_[i].norm();
282  double edge_dist = DistancePointToSegment2D(center, vertices_[i], vertices_[i + 1]);
283  min_dist = std::min(min_dist, std::min(vertex_dist, edge_dist));
284  }
285 
286  // we also need to check the last vertex and the first vertex
287  double vertex_dist = vertices_.back().norm();
288  double edge_dist = DistancePointToSegment2D(center, vertices_.back(), vertices_.front());
289  return std::min(min_dist, std::min(vertex_dist, edge_dist));
290  }
291 
292  private:
293 
294  Point2dContainer vertices_;
295 
296 };
