82  void computeError() {
83  const TebVertexPose *pose1 = static_cast<const TebVertexPose *>(_vertices[0]);
84  const TebVertexPose *pose2 = static_cast<const TebVertexPose *>(_vertices[1]);
85  const TebVertexPose *pose3 = static_cast<const TebVertexPose *>(_vertices[2]);
86  const TebVertexTimeDiff *dt1 = static_cast<const TebVertexTimeDiff *>(_vertices[3]);
87  const TebVertexTimeDiff *dt2 = static_cast<const TebVertexTimeDiff *>(_vertices[4]);
88 
89  const Eigen::Vector2d diff1 = pose2->GetPose().GetPosition() - pose1->GetPose().GetPosition();
90  const Eigen::Vector2d diff2 = pose3->GetPose().GetPosition() - pose2->GetPose().GetPosition();
91 
92  double dist1 = diff1.norm();
93  double dist2 = diff2.norm();
94  const double angle_diff1 = g2o::normalize_theta(pose2->GetPose().GetTheta() - pose1->GetPose().GetTheta());
95  const double angle_diff2 = g2o::normalize_theta(pose3->GetPose().GetTheta() - pose2->GetPose().GetTheta());
96 
97  if (config_param_->trajectory_opt().exact_arc_length()) {
98  if (angle_diff1 != 0) {
99  const double radius = dist1 / (2 * sin(angle_diff1 / 2));
100  dist1 = fabs(angle_diff1 * radius);
101  }
102  if (angle_diff2 != 0) {
103  const double radius = dist2 / (2 * sin(angle_diff2 / 2));
104  dist2 = fabs(angle_diff2 * radius);
105  }
106  }
107 
108  double vel1 = dist1 / dt1->GetDiffTime();
109  double vel2 = dist2 / dt2->GetDiffTime();
110 
111 
112  vel1 *= LogisticSigmoid(100 * (diff1.x() * cos(pose1->GetPose().GetTheta()) + diff1.y() * sin(pose1->GetPose().GetTheta())));
113  vel2 *= LogisticSigmoid(100 * (diff2.x() * cos(pose2->GetPose().GetTheta()) + diff2.y() * sin(pose2->GetPose().GetTheta())));
114 
115  const double acc_lin = (vel2 - vel1) * 2 / (dt1->GetDiffTime() + dt2->GetDiffTime());
116 
117  _error[0] = PenaltyBoundToInterval(acc_lin, config_param_->kinematics_opt().acc_lim_x(),
118  config_param_->optimize_info().penalty_epsilon());
119 
120  const double omega1 = angle_diff1 / dt1->GetDiffTime();
121  const double omega2 = angle_diff2 / dt2->GetDiffTime();
122  const double acc_rot = (omega2 - omega1) * 2 / (dt1->GetDiffTime() + dt2->GetDiffTime());
123 
124  _error[1] = PenaltyBoundToInterval(acc_rot, config_param_->kinematics_opt().acc_lim_theta(),
125  config_param_->optimize_info().penalty_epsilon());
126  }
127 
128  public:
129  EIGEN_MAKE_ALIGNED_OPERATOR_NEW

228 class AccelerationHolonomicEdge : public TebMultiEdgeBase<3, double> {
229  public:

231  AccelerationHolonomicEdge() {

235  void computeError() {
236  const TebVertexPose *pose1 = static_cast<const TebVertexPose *>(_vertices[0]);
237  const TebVertexPose *pose2 = static_cast<const TebVertexPose *>(_vertices[1]);
238  const TebVertexPose *pose3 = static_cast<const TebVertexPose *>(_vertices[2]);
239  const TebVertexTimeDiff *dt1 = static_cast<const TebVertexTimeDiff *>(_vertices[3]);
240  const TebVertexTimeDiff *dt2 = static_cast<const TebVertexTimeDiff *>(_vertices[4]);
241 
242  Eigen::Vector2d diff1 = pose2->GetPose().GetPosition() - pose1->GetPose().GetPosition();
243  Eigen::Vector2d diff2 = pose3->GetPose().GetPosition() - pose2->GetPose().GetPosition();
244 
245  double cos_theta1 = std::cos(pose1->GetPose().GetTheta());
246  double sin_theta1 = std::sin(pose1->GetPose().GetTheta());
247  double cos_theta2 = std::cos(pose2->GetPose().GetTheta());
248  double sin_theta2 = std::sin(pose2->GetPose().GetTheta());
249 
250  double p1_dx = cos_theta1 * diff1.x() + sin_theta1 * diff1.y();
251  double p1_dy = -sin_theta1 * diff1.x() + cos_theta1 * diff1.y();
252 
253  double p2_dx = cos_theta2 * diff2.x() + sin_theta2 * diff2.y();
254  double p2_dy = -sin_theta2 * diff2.x() + cos_theta2 * diff2.y();
255 
256  double vel1_x = p1_dx / dt1->GetDiffTime();
257  double vel1_y = p1_dy / dt1->GetDiffTime();
258  double vel2_x = p2_dx / dt2->GetDiffTime();
259  double vel2_y = p2_dy / dt2->GetDiffTime();
260 
261  double dt12 = dt1->GetDiffTime() + dt2->GetDiffTime();
262 
263  double acc_x = (vel2_x - vel1_x) * 2 / dt12;
264  double acc_y = (vel2_y - vel1_y) * 2 / dt12;
265 
266  _error[0] = PenaltyBoundToInterval(acc_x, config_param_->kinematics_opt().acc_lim_x(),
267  config_param_->optimize_info().penalty_epsilon());
268  _error[1] = PenaltyBoundToInterval(acc_y, config_param_->kinematics_opt().acc_lim_y(),
269  config_param_->optimize_info().penalty_epsilon());
270 
271  double omega1 = g2o::normalize_theta(pose2->GetPose().GetTheta() - pose1->GetPose().GetTheta()) / dt1->GetDiffTime();
272  double omega2 = g2o::normalize_theta(pose3->GetPose().GetTheta() - pose2->GetPose().GetTheta()) / dt2->GetDiffTime();
273  double acc_rot = (omega2 - omega1) * 2 / dt12;
274 
275  _error[2] = PenaltyBoundToInterval(acc_rot, config_param_->kinematics_opt().acc_lim_theta(),
276  config_param_->optimize_info().penalty_epsilon());
277 
278  }
279 
280  public:
281  EIGEN_MAKE_ALIGNED_OPERATOR_NEW
