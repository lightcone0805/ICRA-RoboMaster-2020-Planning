202 class PointObstacle : public Obstacle {
203  public:
204 
205  PointObstacle() : Obstacle(), pos_(Eigen::Vector2d::Zero()) {}
206 
207  PointObstacle(const Eigen::Ref<const Eigen::Vector2d> &position) : Obstacle(), pos_(position) {}
208 
209  PointObstacle(double x, double y) : Obstacle(), pos_(Eigen::Vector2d(x, y)) {}
210 
211  virtual bool CheckCollision(const Eigen::Vector2d &point, double min_dist) const {
212  return GetMinimumDistance(point) < min_dist;
213  }
214 
215  virtual bool CheckLineIntersection(const Eigen::Vector2d &line_start,
216  const Eigen::Vector2d &line_end,
217  double min_dist = 0) const {
218 
219  Eigen::Vector2d a = line_end - line_start;
220  Eigen::Vector2d b = pos_ - line_start;
221 
222  double t = a.dot(b) / a.dot(a);
223  if (t < 0) t = 0;
224  else if (t > 1) t = 1;
225  Eigen::Vector2d nearest_point = line_start + a * t;
226 
227  return CheckCollision(nearest_point, min_dist);
228  }
229 
230  virtual double GetMinimumDistance(const Eigen::Vector2d &position) const {
231  return (position - pos_).norm();
232  }
233 
234  virtual double GetMinimumDistance(const Eigen::Vector2d &line_start, const Eigen::Vector2d &line_end) const {
235  return DistancePointToSegment2D(pos_, line_start, line_end);
236  }
237 
238  virtual double GetMinimumDistance(const Point2dContainer &polygon) const {
239  return DistancePointToPolygon2D(pos_, polygon);
240  }
241 
242  virtual Eigen::Vector2d GetClosestPoint(const Eigen::Vector2d &position) const {
243  return pos_;
244  }
245 
246  virtual const Eigen::Vector2d &GetCentroid() const {
247  return pos_;
248  }
249 
250  virtual std::complex<double> GetCentroidCplx() const {
251  return std::complex<double>(pos_[0], pos_[1]);
252  }
253 
254  const Eigen::Vector2d &Position() const {
255  return pos_;
256  }
257  Eigen::Vector2d &Position() {
258  return pos_;
259  }
260

265  polygon.points.front().z = 0;
266  }

380 class PolygonObstacle : public Obstacle {

382 
383  PolygonObstacle() : Obstacle(), finalized_(false) {
384  centroid_.setConstant(NAN);

388  FinalizePolygon();

391  virtual bool CheckCollision(const Eigen::Vector2d &point, double min_dist) const {
392  if (NoVertices() == 2)
393  return GetMinimumDistance(point) <= min_dist;
394 
395  int i, j;
396  bool c = false;
397  for (i = 0, j = NoVertices() - 1; i < NoVertices(); j = i++) {
398  if (((vertices_.at(i).y() > point.y()) != (vertices_.at(j).y() > point.y())) &&
399  (point.x() < (vertices_.at(j).x() - vertices_.at(i).x()) * (point.y() - vertices_.at(i).y())
400  / (vertices_.at(j).y() - vertices_.at(i).y()) + vertices_.at(i).x())) {
401  c = !c;
402  }
403  }
404  if (c > 0) {
405  return true;
406  }
407 
408  return min_dist == 0 ? false : GetMinimumDistance(point) < min_dist;
409  }
