107  inline unsigned char ComputeCost(unsigned distance_cell) const {
108  unsigned char cost = 0;
109  if (distance_cell == 0)
110  cost = LETHAL_OBSTACLE;
111  else if (distance_cell * resolution_ <= inscribed_radius_)
112  cost = INSCRIBED_INFLATED_OBSTACLE;
113  else {
114  // make sure cost falls off by Euclidean distance
115  double euclidean_distance = distance_cell * resolution_;
116  double factor = exp(-1.0 * weight_ * (euclidean_distance - inscribed_radius_));
117  cost = (unsigned char) ((INSCRIBED_INFLATED_OBSTACLE - 1) * factor);
118  }
119  return cost;
120  }
