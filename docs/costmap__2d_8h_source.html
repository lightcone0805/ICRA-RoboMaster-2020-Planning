390  template<typename ActionType>
391  inline void RaytraceLine(ActionType at, unsigned int x0, unsigned int y0, \
392  unsigned int x1, unsigned int y1, \
393  unsigned int max_length = UINT_MAX) {
394  int dx = x1 - x0;
395  int dy = y1 - y0;
396  unsigned int abs_dx = abs(dx);
397  unsigned int abs_dy = abs(dy);
398  int offset_dx = Sign(dx);
399  int offset_dy = Sign(dy) * size_x_;
400  unsigned int offset = y0 * size_x_ + x0;
401  double dist = hypot(dx, dy);
402  double scale = (dist == 0.0) ? 1.0 : std::min(1.0, max_length / dist);
403  // if x is dominant
404  if (abs_dx >= abs_dy) {
405  int error_y = abs_dx / 2;
406  Bresenham(at, abs_dx, abs_dy, error_y, offset_dx, offset_dy, offset, (unsigned int) (scale * abs_dx));
407  return;
408  }
409  // otherwise y is dominant
410  int error_x = abs_dy / 2;
411  Bresenham(at, abs_dy, abs_dx, error_x, offset_dy, offset_dx, offset, (unsigned int) (scale * abs_dy));
412  }
413 
414  private:
418  template<typename ActionType>
419  inline void Bresenham(ActionType at, unsigned int abs_da, unsigned int abs_db, \
420  int error_b, int offset_a, \
421  int offset_b, unsigned int offset, unsigned int max_length) {
422  unsigned int end = std::min(max_length, abs_da);
423  for (unsigned int i = 0; i < end; ++i) {
424  at(offset);
425  offset += offset_a;
426  error_b += abs_db;
427  if ((unsigned int) error_b >= abs_da) {
428  offset += offset_b;
429  error_b -= abs_da;
430  }
431  }
432  at(offset);
433  }
